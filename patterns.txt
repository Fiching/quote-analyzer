# ============================================================
# CLAUDE INJECTION — Quote Analyzer Pattern File
# ============================================================
# HOW TO USE:
#   - Each PATTERN block defines a fitting type
#   - REQUIRES = fields the AI must find in the quote
#   - IF_MISSING = what to output in the table when a field is absent
#   - SHORTHAND = common abbreviations customers use
#   - ASSUME = safe default if not specified
#   - EXAMPLES = real part numbers from your catalog
#
# HOW TO EDIT:
#   - Add new patterns by copying an existing block
#   - Add new shorthand lines as: customer_term = catalog_term
#   - Add new examples as: PARTNUM = DESCRIPTION
#   - Lines starting with # are comments and are ignored
#   - Save the file and restart Flask — changes load automatically
# ============================================================

PATTERN: nipple
  DESCRIPTION: Short pipe section threaded on both ends
  REQUIRES: size, length (or CLOSE for close nipple), material, schedule
  IF_MISSING_SIZE: NEED MORE INFO — nipple requires a pipe size (e.g. 1/2, 3/4, 1)
  IF_MISSING_LENGTH: NEED MORE INFO — nipple requires a length or CLOSE (e.g. 1/2x3, 1/2 CLOSE)
  IF_MISSING_MATERIAL: NEED MORE INFO — nipple requires material (BLK, GALV, SS, etc.)
  IF_MISSING_SCHEDULE: NEED MORE INFO — nipple requires schedule (XH, STD, S160, etc.)
  ASSUME: if customer says "black nipple" assume BLK XH SMLS A106
  ASSUME: if customer says "galv nipple" assume GALV STL
  SHORTHAND:
    nip = nipple
    close = close nipple (NIPC)
    blk = black (BLK XH SMLS A106)
    galv = galvanized (GALV STL)
    xh = extra heavy
  SIZE_FORMAT: SIZE x LENGTH (e.g. 1/2x3, 3/4x6, 1-1/2x close)
  CATALOG_EXAMPLES:
    BXSNDCL = 1/2XCLOSE BLK XH SMLS A106 NIP
    BXSNDM  = 1/2X3 BLK XH SMLS A106 NIP
    BXSNFM  = 3/4X3 BLK XH SMLS A106 NIP
    GNDJ    = 1/2X1-1/2 GALV STL NIP
    BXSNJM  = 1-1/2X3 BLK XH SMLS A106 NIP


PATTERN: elbow
  DESCRIPTION: Fitting that changes direction of flow — 45 or 90 degrees
  REQUIRES: size, angle (45 or 90), material, schedule/rating
  IF_MISSING_SIZE: NEED MORE INFO — elbow requires a pipe size (e.g. 1/2, 3/4, 1)
  IF_MISSING_ANGLE: NEED MORE INFO — elbow requires angle (45 or 90 degree)
  IF_MISSING_MATERIAL: NEED MORE INFO — elbow requires material (BLK, GALV, SS, FS, etc.)
  IF_MISSING_SCHEDULE: NEED MORE INFO — elbow requires schedule or rating (XH, 3000#, 150#, etc.)
  SHORTHAND:
    ell = elbow
    90 = 90 degree elbow
    45 = 45 degree elbow
    fs 90 = forged steel 3000# threaded 90 elbow
    fs 45 = forged steel 3000# threaded 45 elbow
    blk 90 = black malleable iron 150# 90 elbow
  CATALOG_EXAMPLES:

PATTERN: bushing
  DESCRIPTION: Reducer fitting — ALWAYS has two sizes (large x small)
  REQUIRES: large_size, small_size, material, schedule/rating
  IF_MISSING_LARGE_SIZE: NEED MORE INFO — bushing requires two sizes (e.g. 1x1/2, 3/4x1/2)
  IF_MISSING_SMALL_SIZE: NEED MORE INFO — bushing requires a reduction size (e.g. 1x1/2 — what is the small end?)
  IF_MISSING_MATERIAL: NEED MORE INFO — bushing requires material (BLK, GALV, SS, FS, etc.)
  IF_MISSING_SCHEDULE: NEED MORE INFO — bushing requires schedule or rating (3000#, 150#, etc.)
  SHORTHAND:
    bush = bushing
    hex bush = hex bushing
    fs bush = forged steel 3000# hex threaded bushing
  SIZE_FORMAT: LARGE_SIZE x SMALL_SIZE (e.g. 1x1/2, 2x1-1/2, 3/4x1/4)
  CATALOG_EXAMPLES:
    FSTBDB  = 1/2X1/4 FS 3M-6M HEX THRD BUSH

# ============================================================
# GENERAL RULES
# ============================================================

GENERAL_RULES:
  - If quantity is not specified, default to 1
  - Always output ONLY the pipe-delimited table: QTY|PART_NUMBER|DESCRIPTION
  - If a part cannot be matched AND all info is present, output: QTY|NOT FOUND|customer description
  - If required info is missing for any line item, output: QTY|NEED MORE INFO|what is missing
  - Never invent part numbers — only use exact part numbers from the catalog
  - Ignore any parts starting with SP- (service/special order items)
  - Size format in catalog: fractions use / (e.g. 1/2, 3/4, 1-1/2)
